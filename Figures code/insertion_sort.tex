\begin{algorithm}
\caption{Insertion Sort}
\begin{algorithmic}[1]
\Procedure{InsertionSort}{$arr$, $size$} \Comment{Sort array $arr$ of size $size$}
    \For{$i \gets 1$ \textbf{to} $size-1$} \Comment{Loop through the array}
        \State $current \gets arr[i]$
        \State $temp \gets i - 1$
        \While{$temp \geq 0$ \textbf{and} $arr[temp] > current$} \Comment{Shift elements of $arr[0..i-1]$ that are greater than $current$ to one position ahead}
            \State $arr[temp + 1] \gets arr[temp]$
            \State $temp \gets temp - 1$
        \EndWhile
        \State $arr[temp + 1] \gets current$ \Comment{Place $current$ at its correct position in sorted part}
    \EndFor
\EndProcedure
\end{algorithmic}
\end{algorithm}
